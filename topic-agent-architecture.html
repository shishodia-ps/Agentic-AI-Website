What is an Intelligent Agent?

An intelligent agent is an autonomous entity that observes its environment through sensors and acts upon that environment through actuators to achieve specific goals. Agents can be simple reflex systems or complex reasoning entities with learning capabilities.

Key Insight: The fundamental principle of agent design is the perception-action cycle: agents perceive their environment, process information, make decisions, and execute actions to achieve their objectives.

Agent Architecture Components

The agent architecture shows the flow from perception to action through various processing modules

Types of Intelligent Agents

Simple Reflex Agents

Agents that act based on current percept only, using condition-action rules.

Example: Thermostat that turns heating on/off based on temperature

Model-Based Reflex Agents

Agents that maintain internal state to track aspects of the world not visible in current percept.

Example: Self-driving car tracking other vehicles' positions

Goal-Based Agents

Agents that consider future consequences of actions to achieve specific goals.

Example: Chess program planning moves to achieve checkmate

Utility-Based Agents

Agents that choose actions based on utility functions to maximize expected performance.

Example: Trading bot maximizing profit while minimizing risk

Key Agent Characteristics

Perception

  • • Sensor capabilities and limitations
  • • Data preprocessing and filtering
  • • Feature extraction from raw inputs
  • • Handling noisy or incomplete data

Cognition

  • • Knowledge representation methods
  • • Reasoning and inference mechanisms
  • • Planning and decision-making
  • • Learning from experience

Action

  • • Actuator control and coordination
  • • Action planning and execution
  • • Feedback and adaptation
  • • Safety and constraint handling

Agent Environment Types

Fully Observable vs Partially Observable

Fully Observable: Agent has complete access to environment state (e.g., chess game)

Partially Observable: Agent has limited sensors, must infer hidden state (e.g., self-driving car)

Deterministic vs Stochastic

Deterministic: Next state completely determined by current state and action

Stochastic: Environment has uncertainty, multiple possible outcomes

Episodic vs Sequential

Episodic: Agent experience divided into independent episodes

Sequential: Current decisions affect future opportunities

Static vs Dynamic

Static: Environment remains unchanged while agent deliberates

Dynamic: Environment changes during agent's decision process

# Basic Intelligent Agent Framework
from abc import ABC, abstractmethod
from typing import Any, List, Dict
import random

class Percept:
    def __init__(self, data: Dict[str, Any]):
        self.data = data
        self.timestamp = time.time()

class Action:
    def __init__(self, action_type: str, parameters: Dict[str, Any]):
        self.type = action_type
        self.parameters = parameters

class Environment:
    def __init__(self):
        self.state = {}
        self.agents = []
    
    def add_agent(self, agent):
        self.agents.append(agent)
    
    def get_percept(self, agent):
        return Percept(self.state.copy())
    
    def execute_action(self, action: Action, agent):
        # Update environment based on action
        if action.type == "move":
            self.state['agent_position'] = action.parameters.get('position')
        elif action.type == "sense":
            self.state['last_sensed'] = time.time()

class IntelligentAgent(ABC):
    def __init__(self, name: str):
        self.name = name
        self.knowledge_base = {}
        self.learning_history = []
    
    @abstractmethod
    def perceive(self, percept: Percept) -> Dict[str, Any]:
        """Process sensory input"""
        pass
    
    @abstractmethod
    def think(self, processed_percept: Dict[str, Any]) -> Action:
        """Decision-making process"""
        pass
    
    @abstractmethod
    def act(self, action: Action, environment: Environment):
        """Execute chosen action"""
        pass
    
    def learn(self, experience: Dict[str, Any]):
        """Update knowledge based on experience"""
        self.learning_history.append(experience)
        self.update_knowledge_base(experience)
    
    def update_knowledge_base(self, experience: Dict[str, Any]):
        """Abstract method for knowledge updates"""
        pass

class SimpleReflexAgent(IntelligentAgent):
    def __init__(self, name: str, rules: Dict):
        super().__init__(name)
        self.rules = rules  # condition-action rules
    
    def perceive(self, percept: Percept) -> Dict[str, Any]:
        # Simple perception - just pass through the data
        return percept.data
    
    def think(self, processed_percept: Dict[str, Any]) -> Action:
        # Rule-based decision making
        for condition, action in self.rules.items():
            if self.evaluate_condition(condition, processed_percept):
                return action
        
        # Default action if no rule matches
        return Action("idle", {})
    
    def evaluate_condition(self, condition: str, percept: Dict[str, Any]) -> bool:
        # Simple condition evaluation
        if condition == "low_temperature" and percept.get("temperature", 0) < 20:
            return True
        elif condition == "high_temperature" and percept.get("temperature", 0) > 25:
            return True
        return False
    
    def act(self, action: Action, environment: Environment):
        print(f"{self.name} executing action: {action.type}")
        environment.execute_action(action, self)
    
    def update_knowledge_base(self, experience: Dict[str, Any]):
        # Simple reflex agents don't learn from experience
        pass

class LearningAgent(IntelligentAgent):
    def __init__(self, name: str, learning_rate: float = 0.1):
        super().__init__(name)
        self.learning_rate = learning_rate
        self.q_table = {}  # Q-learning table
    
    def perceive(self, percept: Percept) -> Dict[str, Any]:
        # More sophisticated perception
        processed = percept.data.copy()
        processed['features'] = self.extract_features(percept.data)
        return processed
    
    def extract_features(self, data: Dict[str, Any]) -> List[float]:
        # Extract relevant features from raw percept
        features = []
        if "temperature" in data:
            features.append(data["temperature"] / 100.0)  # Normalize
        if "light" in data:
            features.append(1.0 if data["light"] > 50 else 0.0)  # Binary feature
        return features
    
    def think(self, processed_percept: Dict[str, Any]) -> Action:
        state = self.get_state_representation(processed_percept)
        
        # Choose action using epsilon-greedy policy
        if random.random() < 0.1:  # 10% exploration
            action_type = random.choice(["move", "sense", "idle"])
        else:
            action_type = self.get_best_action(state)
        
        return Action(action_type, {"state": state})
    
    def get_state_representation(self, percept: Dict[str, Any]) -> str:
        # Convert percept to state string
        temp = percept.get("temperature", 0)
        if temp < 20:
            temp_cat = "cold"
        elif temp < 25:
            temp_cat = "comfortable"
        else:
            temp_cat = "hot"
        return f"temp_{temp_cat}"
    
    def get_best_action(self, state: str) -> str:
        if state not in self.q_table:
            self.q_table[state] = {"move": 0, "sense": 0, "idle": 0}
        return max(self.q_table[state], key=self.q_table[state].get)
    
    def update_knowledge_base(self, experience: Dict[str, Any]):
        # Q-learning update
        state = experience["state"]
        action = experience["action"]
        reward = experience["reward"]
        next_state = experience["next_state"]
        
        if state not in self.q_table:
            self.q_table[state] = {"move": 0, "sense": 0, "idle": 0}
        
        # Q-learning formula
        old_q = self.q_table[state][action]
        next_max_q = max(self.q_table.get(next_state, {"move": 0, "sense": 0, "idle": 0}).values())
        new_q = old_q + self.learning_rate * (reward + 0.9 * next_max_q - old_q)
        
        self.q_table[state][action] = new_q

# Example usage
def run_agent_simulation():
    # Create environment
    env = Environment()
    env.state = {"temperature": 22, "light": 60}
    
    # Create simple reflex agent
    reflex_rules = {
        "low_temperature": Action("heat", {"intensity": 0.8}),
        "high_temperature": Action("cool", {"intensity": 0.6})
    }
    reflex_agent = SimpleReflexAgent("Thermostat", reflex_rules)
    
    # Create learning agent
    learning_agent = LearningAgent("AdaptiveController")
    
    # Add agents to environment
    env.add_agent(reflex_agent)
    env.add_agent(learning_agent)
    
    # Run simulation
    for step in range(10):
        print(f"\nStep {step + 1}:")
        
        # Each agent perceives and acts
        for agent in [reflex_agent, learning_agent]:
            percept = env.get_percept(agent)
            processed_percept = agent.perceive(percept)
            action = agent.think(processed_percept)
            agent.act(action, env)
            
            # Simulate learning experience
            experience = {
                "state": agent.get_state_representation(processed_percept) if hasattr(agent, 'get_state_representation') else "unknown",
                "action": action.type,
                "reward": random.uniform(-1, 1),
                "next_state": "next_state"
            }
            agent.learn(experience)

if __name__ == "__main__":
    run_agent_simulation()
