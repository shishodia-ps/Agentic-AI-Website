What is Retrieval-Augmented Generation?

Retrieval-Augmented Generation (RAG) is a powerful technique that combines the strengths of Large Language Models (LLMs) with external knowledge retrieval systems. It enables AI systems to access and utilize up-to-date, domain-specific information that wasn't included in their training data, resulting in more accurate, relevant, and trustworthy responses.

Key Insight: RAG addresses the knowledge cutoff limitations of LLMs by dynamically retrieving relevant information from external sources, enabling AI systems to provide current, factual, and contextually appropriate responses.

RAG Implementation Examples

Basic RAG Implementation 

# Basic RAG Implementation with LangChain
from langchain.document_loaders import WebBaseLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.vectorstores import Chroma
from langchain.embeddings import OpenAIEmbeddings
from langchain.chains import RetrievalQA
from langchain.llms import OpenAI

# Step 1: Load documents
loader = WebBaseLoader("https://example.com/docs")
documents = loader.load()

# Step 2: Split documents into chunks
text_splitter = RecursiveCharacterTextSplitter(
    chunk_size=1000,
    chunk_overlap=200,
    length_function=len
)
texts = text_splitter.split_documents(documents)

# Step 3: Create embeddings and vector store
embeddings = OpenAIEmbeddings()
vectorstore = Chroma.from_documents(texts, embeddings)

# Step 4: Create RAG chain
qa_chain = RetrievalQA.from_chain_type(
    llm=OpenAI(temperature=0),
    chain_type="stuff",
    retriever=vectorstore.as_retriever(
        search_kwargs={"k": 4}  # Retrieve top 4 documents
    ),
    return_source_documents=True
)

# Step 5: Query the system
query = "What are the main features of this product?"
result = qa_chain({"query": query})

print(f"Answer: {result['result']}")
print("\nSource Documents:")
for doc in result['source_documents']:
    print(f"- {doc.metadata.get('source', 'Unknown')}")

Advanced RAG with Multiple Retrievers 

# Advanced RAG with Ensemble Retriever
from langchain.retrievers import EnsembleRetriever
from langchain.retrievers import BM25Retriever
from langchain.schema import BaseRetriever
from langchain.vectorstores import FAISS
from langchain.embeddings import HuggingFaceEmbeddings

# Create multiple retrievers
# 1. Vector-based retriever
embeddings = HuggingFaceEmbeddings(model_name="all-MiniLM-L6-v2")
vectorstore = FAISS.from_documents(texts, embeddings)
vector_retriever = vectorstore.as_retriever(search_kwargs={"k": 3})

# 2. BM25 (keyword-based) retriever
bm25_retriever = BM25Retriever.from_documents(texts)
bm25_retriever.k = 3

# 3. Create ensemble retriever
ensemble_retriever = EnsembleRetriever(
    retrievers=[vector_retriever, bm25_retriever],
    weights=[0.7, 0.3]  # Weight vector search more heavily
)

# Create advanced RAG chain
from langchain.chains import ConversationalRetrievalChain
from langchain.memory import ConversationBufferMemory

memory = ConversationBufferMemory(
    memory_key="chat_history",
    return_messages=True
)

conversational_chain = ConversationalRetrievalChain.from_llm(
    llm=OpenAI(temperature=0.7),
    retriever=ensemble_retriever,
    memory=memory,
    verbose=True
)

# Use the conversational chain
result = conversational_chain({"question": "What are the key benefits?"})
print(f"Answer: {result['answer']}")

RAG with Custom Prompt Templates 

# RAG with Custom Prompt Template
from langchain.prompts import PromptTemplate
from langchain.chains import LLMChain
from langchain.schema.runnable import RunnablePassthrough
from langchain.schema.output_parser import StrOutputParser

# Create custom prompt template
custom_template = """You are an expert assistant specializing in {domain}.
Use the following pieces of context to answer the question at the end.
If you don't know the answer, just say that you don't know, don't try to make up an answer.

Context:
{context}

Question: {question}

Expert Answer:"""

CUSTOM_PROMPT = PromptTemplate(
    template=custom_template,
    input_variables=["context", "question", "domain"]
)

# Create RAG chain with custom prompt
def format_docs(docs):
    return "\n\n".join(doc.page_content for doc in docs)

rag_chain = (
    {"context": ensemble_retriever | format_docs, "question": RunnablePassthrough(), "domain": RunnablePassthrough()}
    | CUSTOM_PROMPT
    | OpenAI(temperature=0)
    | StrOutputParser()
)

# Use the RAG chain
result = rag_chain.invoke({
    "question": "What are the implementation steps?",
    "domain": "software development"
})

print(f"Expert Answer: {result}")

RAG System Evaluation 

# RAG System Evaluation and Metrics
from langchain.evaluation import QAEvalChain
from langchain.schema import Document
import json

# Prepare evaluation dataset
eval_questions = [
    "What is the main purpose of RAG?",
    "How does RAG improve LLM responses?",
    "What are the key components of RAG?",
    "What are the limitations of traditional LLMs?",
    "How is context integrated in RAG?"
]

# Ground truth answers (for evaluation)
ground_truths = [
    "RAG combines LLMs with external knowledge retrieval to provide accurate, up-to-date responses.",
    "RAG improves responses by retrieving relevant context from external sources.",
    "Key components include retrievers, vector stores, embedding models, and LLMs.",
    "Traditional LLMs have knowledge cutoffs and may provide outdated or hallucinated information.",
    "Context is retrieved based on semantic similarity and added to the prompt."
]

# Generate predictions
predictions = []
for question in eval_questions:
    result = qa_chain({"query": question})
    predictions.append(result['result'])

# Create evaluation chain
eval_chain = QAEvalChain.from_llm(
    llm=OpenAI(temperature=0),
    prompt=PromptTemplate(
        template="""Given the following question, ground truth answer, and predicted answer, evaluate the prediction.

Question: {query}
Ground Truth: {answer}
Predicted Answer: {result}

Evaluation (correct/incorrect and explanation):""",
        input_variables=["query", "answer", "result"]
    )
)

# Evaluate predictions
eval_results = eval_chain.evaluate(
    examples=[{"query": q, "answer": a} for q, a in zip(eval_questions, ground_truths)],
    predictions=[{"result": p} for p in predictions],
    question_key="query",
    answer_key="answer",
    prediction_key="result"
)

# Print evaluation results
for i, result in enumerate(eval_results):
    print(f"Question {i+1}: {eval_questions[i]}")
    print(f"Evaluation: {result['text']}")
    print("-" * 50)

RAG Optimization Techniques

Retrieval Optimization

Hybrid Search

Combine semantic and keyword search for better retrieval

Re-ranking

Use cross-encoders to re-rank retrieved documents

Query Expansion

Expand queries with synonyms and related terms

Contextual Compression

Compress retrieved documents to relevant information only

Generation Enhancement

Iterative Refinement

Refine answers through multiple retrieval-generation cycles

Self-Consistency

Generate multiple answers and select the most consistent

Citation Generation

Include citations and source references in responses

Uncertainty Quantification

Indicate confidence levels in generated responses
