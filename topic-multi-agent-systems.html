What are Multi-Agent Systems?

Multi-Agent Systems (MAS) are systems composed of multiple interacting intelligent agents that work together to solve problems that are beyond the capabilities of individual agents. These systems enable complex problem-solving through distributed intelligence, collaboration, and emergent behaviors.

Key Insight: Multi-agent systems leverage the principle that the whole is greater than the sum of its parts—through coordination and specialization, multiple simple agents can solve complex problems that individual agents cannot.

Multi-Agent System Architecture

Multi-agent systems consist of multiple autonomous agents that communicate and coordinate to achieve common goals

Agent Interaction Patterns

Cooperative

Agents work together toward common goals, sharing information and resources.

  • • Shared utility functions
  • • Collaborative problem solving
  • • Resource sharing
  • • Joint planning

Competitive

Agents have conflicting goals and compete for limited resources or outcomes.

  • • Adversarial strategies
  • • Resource competition
  • • Game-theoretic approaches
  • • Zero-sum games

Negotiation-Based

Agents negotiate to reach mutually beneficial agreements or compromises.

  • • Contract net protocols
  • • Auction mechanisms
  • • Bargaining strategies
  • • Mediation processes

Team-Based

Agents form teams with specialized roles to accomplish complex tasks.

  • • Role specialization
  • • Team formation protocols
  • • Distributed task allocation
  • • Coordination mechanisms

# Multi-Agent System Framework
import asyncio
import random
from typing import List, Dict, Any, Optional
from dataclasses import dataclass
from enum import Enum

class MessageType(Enum):
    REQUEST = "request"
    INFORM = "inform"
    PROPOSE = "propose"
    ACCEPT = "accept"
    REJECT = "reject"

@dataclass
class Message:
    sender_id: str
    receiver_id: str
    message_type: MessageType
    content: Dict[str, Any]
    timestamp: float

class MultiAgentEnvironment:
    def __init__(self):
        self.agents = {}
        self.message_queue = asyncio.Queue()
        self.global_state = {}
        self.communication_log = []
    
    def add_agent(self, agent):
        self.agents[agent.agent_id] = agent
        agent.environment = self
    
    async def send_message(self, message: Message):
        await self.message_queue.put(message)
        self.communication_log.append(message)
    
    async def broadcast_message(self, sender_id: str, message_type: MessageType, content: Dict[str, Any]):
        for agent_id in self.agents:
            if agent_id != sender_id:
                message = Message(
                    sender_id=sender_id,
                    receiver_id=agent_id,
                    message_type=message_type,
                    content=content,
                    timestamp=asyncio.get_event_loop().time()
                )
                await self.send_message(message)
    
    async def process_messages(self):
        while True:
            try:
                message = await asyncio.wait_for(self.message_queue.get(), timeout=1.0)
                if message.receiver_id in self.agents:
                    await self.agents[message.receiver_id].receive_message(message)
            except asyncio.TimeoutError:
                break

class MultiAgentBase:
    def __init__(self, agent_id: str, agent_type: str):
        self.agent_id = agent_id
        self.agent_type = agent_type
        self.environment = None
        self.local_state = {}
        self.message_history = []
        self.neighbors = []
    
    async def send_message(self, receiver_id: str, message_type: MessageType, content: Dict[str, Any]):
        if self.environment:
            message = Message(
                sender_id=self.agent_id,
                receiver_id=receiver_id,
                message_type=message_type,
                content=content,
                timestamp=asyncio.get_event_loop().time()
            )
            await self.environment.send_message(message)
    
    async def broadcast(self, message_type: MessageType, content: Dict[str, Any]):
        if self.environment:
            await self.environment.broadcast_message(self.agent_id, message_type, content)
    
    async def receive_message(self, message: Message):
        self.message_history.append(message)
        await self.process_message(message)
    
    @abstractmethod
    async def process_message(self, message: Message):
        pass
    
    @abstractmethod
    async def run(self):
        pass

class CoordinatorAgent(MultiAgentBase):
    def __init__(self, agent_id: str):
        super().__init__(agent_id, "coordinator")
        self.tasks = []
        self.agent_capabilities = {}
    
    async def process_message(self, message: Message):
        if message.message_type == MessageType.INFORM:
            if "capabilities" in message.content:
                self.agent_capabilities[message.sender_id] = message.content["capabilities"]
    
    async def run(self):
        while True:
            # Periodically check for tasks to assign
            if self.tasks:
                task = self.tasks.pop(0)
                suitable_agents = self.find_suitable_agents(task)
                if suitable_agents:
                    chosen_agent = random.choice(suitable_agents)
                    await self.send_message(
                        chosen_agent,
                        MessageType.REQUEST,
                        {"task": task, "deadline": "2 hours"}
                    )
            await asyncio.sleep(5)
    
    def find_suitable_agents(self, task: Dict[str, Any]) -> List[str]:
        suitable = []
        for agent_id, capabilities in self.agent_capabilities.items():
            if task.get("required_skill") in capabilities.get("skills", []):
                suitable.append(agent_id)
        return suitable
    
    async def assign_task(self, task: Dict[str, Any]):
        self.tasks.append(task)

class WorkerAgent(MultiAgentBase):
    def __init__(self, agent_id: str, skills: List[str]):
        super().__init__(agent_id, "worker")
        self.skills = skills
        self.current_task = None
        self.busy = False
    
    async def process_message(self, message: Message):
        if message.message_type == MessageType.REQUEST:
            if not self.busy:
                task = message.content.get("task")
                if task and task.get("required_skill") in self.skills:
                    await self.send_message(
                        message.sender_id,
                        MessageType.PROPOSE,
                        {"task_id": task.get("id"), "estimated_time": "1 hour"}
                    )
    
    async def run(self):
        # Register capabilities with coordinator
        if self.environment:
            coordinator = self.find_coordinator()
            if coordinator:
                await self.send_message(
                    coordinator,
                    MessageType.INFORM,
                    {"capabilities": {"skills": self.skills, "status": "available"}}
                )
        
        while True:
            if self.current_task:
                await self.execute_task(self.current_task)
            await asyncio.sleep(1)
    
    def find_coordinator(self) -> Optional[str]:
        for agent_id, agent in self.environment.agents.items():
            if agent.agent_type == "coordinator":
                return agent_id
        return None
    
    async def execute_task(self, task: Dict[str, Any]):
        self.busy = True
        print(f"Agent {self.agent_id} executing task: {task.get('description', 'Unknown')}")
        
        # Simulate task execution
        await asyncio.sleep(random.uniform(1, 3))
        
        print(f"Agent {self.agent_id} completed task")
        self.current_task = None
        self.busy = False

class MonitorAgent(MultiAgentBase):
    def __init__(self, agent_id: str):
        super().__init__(agent_id, "monitor")
        self.system_metrics = {}
    
    async def process_message(self, message: Message):
        # Monitor system health and performance
        self.system_metrics[message.sender_id] = {
            "last_message": message.timestamp,
            "message_type": message.message_type.value
        }
    
    async def run(self):
        while True:
            # Analyze system performance
            active_agents = len(self.system_metrics)
            print(f"Monitor: {active_agents} agents active")
            
            # Check for failed agents
            current_time = asyncio.get_event_loop().time()
            for agent_id, metrics in self.system_metrics.items():
                if current_time - metrics["last_message"] > 10:
                    print(f"Monitor: Agent {agent_id} appears to be unresponsive")
            
            await asyncio.sleep(10)

# Example MAS Implementation
async def run_multi_agent_system():
    environment = MultiAgentEnvironment()
    
    # Create agents
    coordinator = CoordinatorAgent("coordinator_1")
    worker1 = WorkerAgent("worker_1", ["data_processing", "analysis"])
    worker2 = WorkerAgent("worker_2", ["visualization", "reporting"])
    worker3 = WorkerAgent("worker_3", ["data_processing", "machine_learning"])
    monitor = MonitorAgent("monitor_1")
    
    # Add agents to environment
    environment.add_agent(coordinator)
    environment.add_agent(worker1)
    environment.add_agent(worker2)
    environment.add_agent(worker3)
    environment.add_agent(monitor)
    
    # Create tasks
    tasks = [
        {"id": 1, "description": "Process sensor data", "required_skill": "data_processing"},
        {"id": 2, "description": "Generate report", "required_skill": "reporting"},
        {"id": 3, "description": "Train ML model", "required_skill": "machine_learning"}
    ]
    
    # Assign tasks
    for task in tasks:
        await coordinator.assign_task(task)
    
    # Run the system
    tasks = [
        coordinator.run(),
        worker1.run(),
        worker2.run(),
        worker3.run(),
        monitor.run(),
        environment.process_messages()
    ]
    
    await asyncio.gather(*tasks)

# Run the example
if __name__ == "__main__":
    asyncio.run(run_multi_agent_system())
